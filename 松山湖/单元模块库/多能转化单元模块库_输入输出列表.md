### 关键参数（来自中芬项目代码库）

|:------|:--:|:-----|

### 出力曲线模型
```python
if v < 2.5:
    P = 0
elif v < 9:
    P = P_r × (v³ - 2.5³) / (9³ - 2.5³)
elif v < 25:
    P = P_r
else:
    P = 0
```

## 使用示例

```python
from energy_units_library import (
    CHPUnit, CHPParams,
    WindTurbineUnit, WindTurbineParams,
    PhotovoltaicUnit, PhotovoltaicParams,
    BatteryStorageUnit, BatteryStorageParams
)
import oemof.solph as solph
import pandas as pd

# 创建能源系统
time_index = pd.date_range("2024-01-01", periods=24, freq="H")
es = solph.EnergySystem(timeindex=time_index, infer_last_interval=True)

# 创建母线
ele_bus = solph.Bus(label="electricity bus")
heat_bus = solph.Bus(label="heat bus")
gas_bus = solph.Bus(label="gas bus")
es.add(ele_bus, heat_bus, gas_bus)

# 创建CHP（使用中芬项目默认参数）
chp_unit = CHPUnit()  # 使用默认参数
chp = chp_unit.create_component(gas_bus, ele_bus, heat_bus)
es.add(chp)

# 创建风机
wt_unit = WindTurbineUnit(WindTurbineParams(nominal_power=500))
wind_speed = [5, 6, 7, 8, 9, 10, ...]  # 24小时风速数据
wt_output = wt_unit.calc_output_curve(wind_speed)
wt = wt_unit.create_component(ele_bus, wt_output)
es.add(wt)

# 创建蓄电池
battery_unit = BatteryStorageUnit(BatteryStorageParams(
    nominal_power=200,
    nominal_capacity=400
))
battery = battery_unit.create_component(ele_bus)
es.add(battery)
```
